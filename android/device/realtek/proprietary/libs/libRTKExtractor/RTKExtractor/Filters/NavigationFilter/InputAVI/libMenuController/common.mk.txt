INCLUDE(../../../../../include/config.txt)
SET(LIB_DIR ../../../../../lib)


IF(TARGET_BOARD STREQUAL YES)
SET(CPU MIPS)
ELSE(TARGET_BOARD STREQUAL YES)
SET(CPU X86)
ENDIF(TARGET_BOARD STREQUAL YES)


SET(LARGE_FILE_SUPPORT 1)

IF(LARGE_FILE_SUPPORT STREQUAL 1)
SET(LFS " -DLARGE_FILE_SUPPORT -D_LARGEFILE_SOURCE -D_FILE_OFFSET_BITS=64")
message("warning USING 64 BIT FILE I/O")
ELSE(LARGE_FILE_SUPPORT STREQUAL 1)
SET(LFS " -DSMALL_FILE_SUPPORT")
message("warning USING 32 BIT FILE I/O. File size limited to 2GByte")
ENDIF(LARGE_FILE_SUPPORT STREQUAL 1)


SET(PLATFORM " -DLINUX -DDIVXINT_NATIVE_64BIT")

SET(CFLAGS ${CFLAGS} $(DEBUG) $(PLATFORM) $(OPTIMIZE) $(CPU_OPTIMIZE) $(INCLUDE_DIRS) $(LFS))

SET(LDFLAGS ${LDFLAGS} ${OPTIMIZE})
